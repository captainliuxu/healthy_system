        self.code = code
        self.message = message


async def business_exception_handler(request: Request, exc: BusinessException):
    return JSONResponse(
        status_code=400,
        content={"code": exc.code, "message": exc.message, "data": None},
    )


async def generic_exception_handler(request: Request, exc: Exception):

from sqlalchemy.orm import Session
from app.db.models.profile import Profile


class ProfileRepository:
    def create(self, db: Session, obj_in: dict) -> Profile:
        db_obj = Profile(**obj_in)
        db.add(db_obj)
        db.commit()
        db.refresh(db_obj)
        return db_obj

    def get_by_id(self, db: Session, profile_id: int) -> Profile | None:
        return db.query(Profile).filter(Profile.id == profile_id).first()

    def list(self, db: Session, skip: int = 0, limit: int = 20) -> list[Profile]:
        return db.query(Profile).offset(skip).limit(limit).all()

    def update(self, db: Session, db_obj: Profile, obj_in: dict) -> Profile:
        for field, value in obj_in.items():
            setattr(db_obj, field, value)
        db.add(db_obj)
        db.commit()
        db.refresh(db_obj)
        return db_obj

    def delete(self, db: Session, db_obj: Profile):
        db.delete(db_obj)
        db.commit()

app/services/profile_service.py

from sqlalchemy.orm import Session
from app.core.exception import BusinessException
from app.repositories.profile_repo import ProfileRepository
from app.schemas.profile import ProfileCreate, ProfileUpdate


class ProfileService:
    def __init__(self):
        self.repo = ProfileRepository()

    def create_profile(self, db: Session, obj_in: ProfileCreate):
        return self.repo.create(db, obj_in.model_dump())

    def get_profile(self, db: Session, profile_id: int):
        profile = self.repo.get_by_id(db, profile_id)
        if not profile:
            raise BusinessException(40401, "老人档案不存在")
        return profile

    def list_profiles(self, db: Session, skip: int = 0, limit: int = 20):
        return self.repo.list(db, skip=skip, limit=limit)

    def update_profile(self, db: Session, profile_id: int, obj_in: ProfileUpdate):
        profile = self.get_profile(db, profile_id)
        return self.repo.update(db, profile, obj_in.model_dump(exclude_unset=True))

    def delete_profile(self, db: Session, profile_id: int):
        profile = self.get_profile(db, profile_id)
        self.repo.delete(db, profile)

    def create_record(self, db: Session, obj_in: RecordCreate):
        profile = self.profile_repo.get_by_id(db, obj_in.profile_id)
        if not profile:
            raise BusinessException(40401, "老人档案不存在")
        payload = obj_in.model_dump()
        if payload["recorded_at"] is None:
            payload.pop("recorded_at")
        return self.record_repo.create(db, payload)

    def list_records_by_profile(self, db: Session, profile_id: int):
        profile = self.profile_repo.get_by_id(db, profile_id)
        if not profile:
            raise BusinessException(40401, "老人档案不存在")
        return self.record_repo.list_by_profile(db, profile_id)

from app.core.config import settings


class BaseLLMService:
    def generate(self, messages: list[dict]) -> str:
        raise NotImplementedError


class MockLLMService(BaseLLMService):
    def generate(self, messages: list[dict]) -> str:
        last_user_message = ""
        for item in reversed(messages):
            if item["role"] == "user":
                last_user_message = item["content"]
                break
        return f"这是模拟回复：我已经收到你的问题“{last_user_message}”，建议先保持规律作息，并结合最近健康数据继续观察。"


class DashScopeLLMService(BaseLLMService):
    def generate(self, messages: list[dict]) -> str:
        if not settings.DASHSCOPE_API_KEY:
            return MockLLMService().generate(messages)
        return MockLLMService().generate(messages)


def get_llm_service() -> BaseLLMService:
    if settings.LLM_PROVIDER == "mock":
        return MockLLMService()
    return DashScopeLLMService()

from sqlalchemy.orm import Session
from app.core.config import settings
from app.repositories.message_repo import MessageRepository
from app.repositories.profile_repo import ProfileRepository
from app.repositories.record_repo import RecordRepository


class MemoryService:
    def __init__(self):
        self.message_repo = MessageRepository()
        self.profile_repo = ProfileRepository()
        self.record_repo = RecordRepository()

    def build_messages(self, db: Session, profile_id: int, conversation_id: int, user_message: str) -> list[dict]:
        profile = self.profile_repo.get_by_id(db, profile_id)
        history = self.message_repo.list_recent(db, conversation_id=conversation_id, limit=settings.CHAT_MEMORY_LIMIT)
        recent_bp = self.record_repo.latest_by_type(db, profile_id, "blood_pressure")
        recent_hr = self.record_repo.latest_by_type(db, profile_id, "heart_rate")

        messages = [{"role": "system", "content": "你是一个面向老人健康场景的陪护助手，需要给出温和、简洁、可执行的建议。"}]

        if profile:
            messages.append({
                "role": "system",
                "content": f"用户姓名：{profile.name}；年龄：{profile.age or '未知'}；慢病：{profile.chronic_disease or '无'}；服药备注：{profile.medicine_note or '无'}。",
            })

        if recent_bp:
            messages.append({"role": "system", "content": f"最近血压记录：{recent_bp.value_json}"})
        if recent_hr:
            messages.append({"role": "system", "content": f"最近心率记录：{recent_hr.value_json}"})

        messages.extend({"role": item.role, "content": item.content} for item in history)
        messages.append({"role": "user", "content": user_message})
        return messages

    def build_proactive_prompt(self, profile_name: str, trigger_type: str, reason: str) -> list[dict]:
        return [
            {"role": "system", "content": "你是一个老人健康陪护助手，需要生成一句主动关怀消息，长度控制在50字以内，语气自然。"},
            {"role": "user", "content": f"请给{profile_name}生成一条主动消息。触发类型：{trigger_type}。触发原因：{reason}。"},
        ]

        profile = self.profile_repo.get_by_id(db, req.profile_id)
        if not profile:
            raise BusinessException(40401, "老人档案不存在")

        if req.conversation_id:
            conversation = self.conversation_repo.get_by_id(db, req.conversation_id)
            if not conversation:
                raise BusinessException(40402, "会话不存在")
        else:
            conversation = self.conversation_repo.create(db, {"profile_id": req.profile_id, "title": f"{profile.name}的会话"})

        user_message = self.message_repo.create(db, {"conversation_id": conversation.id, "role": "user", "content": req.message, "message_type": "chat"})
        messages = self.memory_service.build_messages(db, req.profile_id, conversation.id, req.message)
        reply = self.llm_service.generate(messages)
        assistant_message = self.message_repo.create(db, {"conversation_id": conversation.id, "role": "assistant", "content": reply, "message_type": "chat"})

        return {
            "conversation_id": conversation.id,
            "user_message_id": user_message.id,
            "assistant_message_id": assistant_message.id,
            "reply": reply,
            "replied_at": datetime.now(UTC),
        }

app/services/trigger_service.py

import json
from datetime import datetime
from sqlalchemy.orm import Session
from app.repositories.proactive_repo import ProactiveRepository
from app.repositories.record_repo import RecordRepository


class TriggerService:
    def __init__(self):
        self.record_repo = RecordRepository()
        self.proactive_repo = ProactiveRepository()

    def check_trigger(self, db: Session, profile_id: int) -> dict:
        window_result = self._check_time_window(db, profile_id)
        if window_result["should_trigger"]:
            return window_result
        bp_result = self._check_blood_pressure(db, profile_id)
        if bp_result["should_trigger"]:
            return bp_result
        return {"should_trigger": False, "trigger_type": "none", "reason": "当前没有满足主动触发条件"}

    def _check_time_window(self, db: Session, profile_id: int) -> dict:
        now_str = datetime.now().strftime("%H:%M")
        windows = self.proactive_repo.list_windows(db, profile_id)
        for window in windows:
            if window.enabled and window.start_time <= now_str <= window.end_time:
                return {"should_trigger": True, "trigger_type": f"window_{window.scene}", "reason": f"当前时间 {now_str} 落在 {window.scene} 主动关怀时间窗内"}
        return {"should_trigger": False, "trigger_type": "none", "reason": ""}

    def _check_blood_pressure(self, db: Session, profile_id: int) -> dict:
        latest_bp = self.record_repo.latest_by_type(db, profile_id, "blood_pressure")
        if not latest_bp:
            return {"should_trigger": False, "trigger_type": "none", "reason": ""}
        try:
            data = json.loads(latest_bp.value_json)
        except json.JSONDecodeError:
            return {"should_trigger": False, "trigger_type": "none", "reason": ""}
        systolic = data.get("systolic")
        diastolic = data.get("diastolic")
        if systolic and diastolic and (systolic >= 140 or diastolic >= 90):
            return {"should_trigger": True, "trigger_type": "health_alert", "reason": f"最近一次血压记录偏高：{systolic}/{diastolic} mmHg"}
        return {"should_trigger": False, "trigger_type": "none", "reason": ""}

from sqlalchemy.orm import Session
from app.core.exception import BusinessException
from app.repositories.conversation_repo import ConversationRepository
from app.repositories.message_repo import MessageRepository
from app.repositories.profile_repo import ProfileRepository
from app.repositories.proactive_repo import ProactiveRepository
from app.services.llm_service import get_llm_service
from app.services.memory_service import MemoryService
from app.services.trigger_service import TriggerService


class ProactiveService:
    def __init__(self):
        self.profile_repo = ProfileRepository()
        self.conversation_repo = ConversationRepository()
        self.message_repo = MessageRepository()
        self.proactive_repo = ProactiveRepository()
        self.trigger_service = TriggerService()
        self.memory_service = MemoryService()
        self.llm_service = get_llm_service()

    def create_window(self, db: Session, payload: dict):
        profile = self.profile_repo.get_by_id(db, payload["profile_id"])
        if not profile:
            raise BusinessException(40401, "老人档案不存在")
        return self.proactive_repo.create_window(db, payload)

    def list_windows(self, db: Session, profile_id: int):
        return self.proactive_repo.list_windows(db, profile_id)

    def list_logs(self, db: Session, profile_id: int):
        return self.proactive_repo.list_logs(db, profile_id)

    def trigger_for_profile(self, db: Session, profile_id: int) -> dict:
        profile = self.profile_repo.get_by_id(db, profile_id)
        if not profile:
            raise BusinessException(40401, "老人档案不存在")

        trigger_result = self.trigger_service.check_trigger(db, profile_id)
        if not trigger_result["should_trigger"]:
            return {"triggered": False, "reason": trigger_result["reason"], "message": None}

        conversations = self.conversation_repo.list_by_profile(db, profile_id)
        conversation = conversations[0] if conversations else self.conversation_repo.create(db, {"profile_id": profile_id, "title": f"{profile.name}的主动关怀会话"})

        prompt = self.memory_service.build_proactive_prompt(profile.name, trigger_result["trigger_type"], trigger_result["reason"])
        content = self.llm_service.generate(prompt)
        self.message_repo.create(db, {"conversation_id": conversation.id, "role": "assistant", "content": content, "message_type": "proactive"})
        log = self.proactive_repo.create_log(db, {"profile_id": profile_id, "trigger_type": trigger_result["trigger_type"], "trigger_reason": trigger_result["reason"], "message_content": content, "status": "sent"})
        return {"triggered": True, "conversation_id": conversation.id, "content": content, "log_id": log.id, "trigger_type": trigger_result["trigger_type"], "reason": trigger_result["reason"]}

from collections import defaultdict
from fastapi import WebSocket


class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[int, list[WebSocket]] = defaultdict(list)

    async def connect(self, profile_id: int, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[profile_id].append(websocket)

    def disconnect(self, profile_id: int, websocket: WebSocket):
        if profile_id in self.active_connections and websocket in self.active_connections[profile_id]:
            self.active_connections[profile_id].remove(websocket)
            if not self.active_connections[profile_id]:
                self.active_connections.pop(profile_id, None)

    async def send_to_profile(self, profile_id: int, payload: dict):
        for connection in self.active_connections.get(profile_id, []):
            await connection.send_json(payload)


manager = ConnectionManager()

from apscheduler.schedulers.background import BackgroundScheduler
from app.core.config import settings
from app.db.session import SessionLocal
from app.repositories.profile_repo import ProfileRepository
from app.services.proactive_service import ProactiveService

scheduler = BackgroundScheduler()


def proactive_job():
    db = SessionLocal()
    try:
        profile_repo = ProfileRepository()
        proactive_service = ProactiveService()
        profiles = profile_repo.list(db, skip=0, limit=1000)
        for profile in profiles:
            proactive_service.trigger_for_profile(db, profile.id)
    finally:
        db.close()


def start_scheduler():
    if not settings.PROACTIVE_SCHEDULER_ENABLED:
        return
    if scheduler.running:
        return
    scheduler.add_job(proactive_job, "interval", minutes=settings.PROACTIVE_SCHEDULER_MINUTES, id="proactive_job", replace_existing=True)
    scheduler.start()


def shutdown_scheduler():
    if scheduler.running:
        scheduler.shutdown()
