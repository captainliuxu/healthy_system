5. core/response.py

统一响应格式是非常有用的。

def success(data=None, message="success", code=0):
    return {
        "code": code,
        "message": message,
        "data": data
    }

def fail(message="fail", code=1, data=None):
    return {
        "code": code,
        "message": message,
        "data": data
    }

前端以后就固定按：

  • code
  • message
  • data

去处理。

class ProfileCreate(BaseModel):
    name: str = Field(..., min_length=1, max_length=50)
    age: int = Field(..., ge=0, le=120)
    gender: str = Field(..., min_length=1, max_length=10)

    emergency_contact: Optional[str] = None

class ProfileItem(ProfileCreate):
    id: int

10. services/chat_service.py 

import dashscope
from app.core.config import DASHSCOPE_API_KEY

dashscope.api_key = DASHSCOPE_API_KEY

def get_chat_reply(message: str) -> str:
    if not DASHSCOPE_API_KEY:
        return "后端未读取到 DASHSCOPE_API_KEY，请检查 .env"

    response = dashscope.Generation.call(
        model="qwen-turbo",
        messages=[
            {"role": "system", "content": "你是一个慢病健康管理助手，回答简洁、温和"},
            {"role": "user", "content": message}
        ]
    )

    return response["output"]["text"]

这里为什么这么写

因为这一步的目标是先把模型调用从路由层拿走。
以后要加：

  • prompt 模板
  • 用户上下文
  • 会话历史
  • 重试机制
  • 限流
    都可以继续在 service 层扩展。

11. services/profile_service.py 

from app.data.fake_store import profiles

def create_profile(data):
    new_profile = {
        "id": len(profiles) + 1,
        **data.model_dump()
    }
    profiles.append(new_profile)
    return new_profile

def list_profiles():
    return profiles

def get_profile(profile_id: int):
    for item in profiles:
        if item["id"] == profile_id:
            return item
    return None

12. services/record_service.py 

from datetime import datetime
from app.data.fake_store import records

def create_record(data):
    new_record = {
        "id": len(records) + 1,
        **data.model_dump(),
        "created_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }
    records.append(new_record)
    return new_record

def list_records():
    return records

def list_records_by_profile(profile_id: int):
    return [item for item in records if item["profile_id"] == profile_id]

16. api/routes/profile.py 

from fastapi import APIRouter, HTTPException
from app.schemas.profile import ProfileCreate
from app.services.profile_service import create_profile, list_profiles, get_profile
from app.core.response import success

router = APIRouter(prefix="/api/profiles", tags=["profiles"])

@router.post("")
def add_profile(data: ProfileCreate):
    result = create_profile(data)
    return success(data=result, message="档案创建成功")

@router.get("")
def get_profiles():
    return success(data=list_profiles())

@router.get("/{profile_id}")
def read_profile(profile_id: int):
    profile = get_profile(profile_id)
    if not profile:
        raise HTTPException(status_code=404, detail="档案不存在")
    return success(data=profile)

18. app/main.py 

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

from app.core.config import APP_NAME
from app.api.routes.health import router as health_router
from app.api.routes.system import router as system_router
from app.api.routes.chat import router as chat_router
from app.api.routes.profile import router as profile_router
from app.api.routes.record import router as record_router

app = FastAPI(title=APP_NAME)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

app.include_router(health_router)
app.include_router(system_router)
app.include_router(chat_router)
app.include_router(profile_router)
app.include_router(record_router)
